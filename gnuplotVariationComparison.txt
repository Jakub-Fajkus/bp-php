clear
reset
unset key
set term png size 1280, 960
set output 'variationComparisonGraph.png'

# Make the x axis labels easier to read.
set xtics rotate out
# Select histogram data
set style data histogram
# Give the bars a plain fill pattern, and draw a solid line around them.
set style fill solid border

set title "GA config comparison"

set grid ytics
set yrange [0:]


set style histogram clustered
set xlabel "Measure"
set ylabel "Fitness"
set key auto columnheader

plot for [COL=2:10] 'variationComparison.data' using COL:xticlabels(1) title columnheader
